def gini(gdp):

    gdp = gdp.sort_values(ascending=True)
    n = len(gdp)

    # Make top of left side fraction
    top = 0
    for i in range(1, n+1):
        print(i)
        top = top + gdp.iloc[i] * i

    top = top * 2

    # Make bottom of left side fraction
    bottom = gdp.sum() * len(gdp)

    # make correction
    correction = (n + 1) / n

    # Bring together
    return top / bottom - correction

def gini2(gdp):

    gdp = gdp.sort_values(ascending=True)
    n = len(gdp)

    # Make top of left side fraction
    top = 0
    for i in range(1, n+1):
        top = top + gdp[i-1] * i

    top = top * 2

    # Make bottom of left side fraction
    bottom = gdp.sum() * len(gdp)

    # make correction
    correction = (n + 1) / n

    # Bring together
    return top / bottom - correction
